## TypeScript Interfaces

### Core Data Types

```typescript
// Property Interface
interface Property {
  id: string | number;
  title: string;
  slug: string;
  description: string;
  price: number;
  priceDisplay: string;
  pricePeriod: 'mo' | 'year' | '';
  status: 'For Sale' | 'For Rent' | 'Sold';
  featured: boolean;
  location: {
    address: string;
    city: string;
    state: string;
    country: string;
    coordinates: {
      lat: number;
      lng: number;
    };
  };
  specs: {
    beds: number;
    baths: number;
    sqft: number;
    yearBuilt?: number;
  };
  images: string[];
  amenities: string[];
  agent: {
    id: string;
    name: string;
    avatar: string;
    phone: string;
    email: string;
  };
  createdAt: string;
  updatedAt: string;
}

// Search Filters Interface
interface SearchFilters {
  listingStatus: 'All' | 'Buy' | 'Rent';
  propertyType: string[];
  priceMin: number;
  priceMax: number;
  beds: number | null;
  baths: number | null;
  location: string;
  sqftMin: number;
  sqftMax: number;
  yearBuiltMin: number;
  yearBuiltMax: number;
  amenities: string[];
}

// Pagination Interface
interface Pagination {
  currentPage: number;
  totalPages: number;
  totalItems: number;
  itemsPerPage: number;
  showingFrom: number;
  showingTo: number;
}

// API Response Interface
interface PropertyListResponse {
  data: Property[];
  pagination: Pagination;
  filters: SearchFilters;
}
```
